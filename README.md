<h3>🚀 Como usar este projeto</h3>

    # Clone este repósitório
    git clone https://github.com/leandro-wrf/next-level-week-01

    # Execute este projeto
    # backend
    cd next-level-week-01 && yarn
    yarn knex migrate:latest
    yarn seed:run
    yarn dev

    # frontend
    cd next-level-week-01 && yarn 
    yarn start

    # mobile
    cd next-level-week-01 && yarn
    expo start
